470 void Function<R(As...)>::swap(Function &other) noexcept {
471  if (this == &other || (!vptr_ && !other.vptr_)) {
472  return;
473  }
474 
475  if (vptr_ == other.vptr_ && !is_ptr_ && !other.is_ptr_) {
476  vptr_->swap(&storage_, &other.storage_);
477  } else if (!vptr_) {
478  is_ptr_ = other.is_ptr_;
479 
480  if (is_ptr_) {
481  as_ptr() = other.as_ptr();
482  } else {
483  other.vptr_->move(&storage_, &other.storage_);
484  other.vptr_->destroy(&other.storage_);
485  }
486  } else if (!other.vptr_) {
487  other.is_ptr_ = is_ptr_;
488 
489  if (is_ptr_) {
490  other.as_ptr() = as_ptr();
491  } else {
492  vptr_->move(&other.storage_, &storage_);
493  vptr_->destroy(&storage_);
494  }
495  } else if (is_ptr_) {
496  if (other.is_ptr_) {
497  std::swap(as_ptr(), other.as_ptr());
498  } else {
499  Storage temp;
500  other.vptr_->move(&temp, &other.storage_);
501  other.vptr_->destroy(&other.storage_);
502 
503  other.is_ptr_ = true;
504  other.as_ptr() = as_ptr();
505 
506  is_ptr_ = false;
507  other.vptr_->move(&storage_, &temp);
508  other.vptr_->destroy(&temp);
509  }
510  } else {
511  if (other.is_ptr_) {
512  Storage temp;
513  vptr_->move(&temp, &storage_);
514  vptr_->destroy(&storage_);
515 
516  is_ptr_ = true;
517  as_ptr() = other.as_ptr();
518 
519  other.is_ptr_ = false;
520  vptr_->move(&other.storage_, &temp);
521  vptr_->destroy(&temp);
522  } else {

525  vptr_->destroy(&storage_);
526

528  other.vptr_->destroy(&other.storage_);

532  }

571 void Function<R(As...)>::construct(Ts &&...ts) {
572  static_assert(
573  std::is_constructible_v<std::decay_t<F>, Ts...>,
574  "std::decay_t<F> must be constructible from (Ts...)"
575  );
576 
577  using Obj = std::decay_t<F>;
578 
579  assert(!vptr_);
580 
581  vptr_ = &detail::get_vtbl<Obj, R, As...>();
582 
583  if constexpr (sizeof(Obj) <= sizeof(Storage) && alignof(Storage) % alignof(Obj) == 0) {
584  is_ptr_ = false;
585  new (&storage_) Obj(std::forward<Ts>(ts)...);
586  } else {

588 
589  is_ptr_ = true;
